# streamlit_gpt_api_ex
streamlit연습

# 가상환경 만들기
```
conda create -n gpt_env python=3.12
conda activate gpt_env
```

# 가상환경 jupyter 노트북 등록하기
```
pip install ipykernel
python -m ipykernel install --user --name gpt_env
```
